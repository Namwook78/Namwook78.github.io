function solution(num1, num2) {
  let arr = [];
  let count = 0;
  for (let index = 0; index <= num2; index++) {
    arr.push(true);
  }
  if (num1 === 0 || num1 === 1) {
    arr.splice(0, 2, false, false);
  }

  for (var m = 2; m <= num2; m++) {
    if (arr[m]) {
      for (var n = 2; n <= num2 / m; n++) {
        arr[m * n] = false;
      }
    }
  }
  for (var j = num1 + 1; j <= num2; j++) {
    if (arr[j]) {
      count++;
    }
  }
  return count;
}
